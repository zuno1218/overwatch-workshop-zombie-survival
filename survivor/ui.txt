rule("스태미나 UI 생성")
{
	event
	{
		Ongoing - Each Player;
		All;
		All;
	}

	condition
	{
		Global.gameProgress == True;
		Hero Of(Event Player) == Hero(솔저: 76);
		Event Player.stamina != 100;
	}

	action
	{
		Create Progress Bar In-World Text(Event Player, Event Player.stamina, Custom String("스태미나"), World Vector Of(Vector(2,
			1 - Sine From Degrees(Vertical Facing Angle Of(Event Player)) * Event Player.cameraPosition[2], Cosine From Degrees(
			Vertical Facing Angle Of(Event Player)) * 2), Event Player, Rotation And Translation), 0.100, Do Not Clip,
			Event Player.staminaColor, Color(흰색), Visible To Position Values and Color, Default Visibility);
		Event Player.staminaProgressBar = Last Text ID;
	}
}
