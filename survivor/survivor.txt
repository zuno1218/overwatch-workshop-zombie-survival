rule("솔저 기본 설정")
{
	event
	{
		Ongoing - Each Player;
		All;
		All;
	}

	condition
	{
		Global.gameProgress == True;
		Hero Of(Event Player) == Hero(솔저: 76);
	}

	action
	{
		Start Damage Modification(Event Player, Remove From Array(Players On Hero(Hero(솔저: 76), All Teams), Event Player), 0,
			Receivers Damagers and Damage Percent);
		Event Player.damageMod = Last Damage Modification ID;
		Start Forcing Player Outlines(Event Player, Remove From Array(Players On Hero(Hero(솔저: 76), All Teams), Event Player), False,
			Color(흰색), Default);
	}
}

rule("데미지")
{
	event
	{
		Player Dealt Damage;
		All;
		All;
	}

	condition
	{
		Global.gameProgress == True;
		Hero Of(Victim) == Hero(로드호그);
		Event Damage < 100;
	}

	action
	{
		Global.zombieKingHealth -= Event Damage;
	}
}

rule("넉백")
{
	event
	{
		Player Dealt Damage;
		All;
		All;
	}

	condition
	{
		Global.gameProgress == True;
		Hero Of(Victim) != Hero(솔저: 76);
		(Event Damage < 5.500 || Hero Of(Victim) == Hero(정크랫)) == True;
		Has Status(Victim, Knocked Down) == False;
		Global.ultra == False;
	}

	action
	{
		Apply Impulse(Victim, Vector(X Component Of(Victim) + -1 * X Component Of(Direction Towards(Victim, Event Player)), Y Component Of(
			Victim), Z Component Of(Victim) + -1 * Z Component Of(Direction Towards(Victim, Event Player))), 5, To World,
			Cancel Contrary Motion);
	}
}
