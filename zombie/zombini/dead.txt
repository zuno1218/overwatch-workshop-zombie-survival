rule("사망")
{
	event
	{
		Player Died;
		All;
		All;
	}

	condition
	{
		Global.gameProgress == True;
		Hero Of(Event Player) == Hero(정크랫);
	}

	action
	{
		Event Player.zombieRespawnProgress = 0;
		If(Event Was Environment == True);
			Respawn(Event Player);
			Wait(0.500, Ignore Condition);
		End;
		Resurrect(Event Player);
		Set Player Health(Event Player, 1);
		Set Facing(Event Player, Vector(0, -1, 0), To Player);
		Set Status(Event Player, Null, Unkillable, 999);
		Set Status(Event Player, Null, Knocked Down, 999);
		Start Accelerating(Event Player, Vector(0, 2, 2), 70, 999, To Player, Direction Rate and Max Speed);
		Wait(0.250, Ignore Condition);
		Stop Accelerating(Event Player);
		Create Progress Bar In-World Text(Event Player, Event Player.zombieRespawnProgress, Custom String("부활"), World Vector Of(Vector(
			Event Player.cameraPosition[0], 0, -1.500), Event Player, Rotation And Translation), 0.900, Do Not Clip, Color(Yellow), Color(
			흰색), Visible To Position and Values, Default Visibility);
		Event Player.zombieRespawnProgressBar = Last Text ID;
		Chase Player Variable Over Time(Event Player, zombieRespawnProgress, 100, 5, Destination and Duration);
		Wait(5, Ignore Condition);
		Heal(Event Player, Null, 1000);
		Clear Status(Event Player, Unkillable);
		Clear Status(Event Player, Knocked Down);
		Stop Chasing Player Variable(Event Player, zombieRespawnProgress);
		Destroy Progress Bar In-World Text(Event Player.zombieRespawnProgressBar);
	}
}

rule("사망 기절")
{
	event
	{
		Ongoing - Each Player;
		All;
		All;
	}

	condition
	{
		Global.gameProgress == True;
		Hero Of(Event Player) == Hero(정크랫);
		Has Status(Event Player, Unkillable) == True;
		Has Status(Event Player, Knocked Down) == False;
	}

	action
	{
		Wait(0.100, Ignore Condition);
		Set Facing(Event Player, Vector(0, -1, 0), To Player);
		Set Status(Event Player, Null, Knocked Down, 999);
	}
}
