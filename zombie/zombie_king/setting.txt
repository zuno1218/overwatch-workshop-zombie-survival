rule("로드호그 기본 설정")
{
	event
	{
		Ongoing - Each Player;
		All;
		All;
	}

	condition
	{
		Global.gameProgress == True;
		Hero Of(Event Player) == Hero(로드호그);
	}

	action
	{
		Event Player.cameraPosition[0] = -1.100;
		Event Player.cameraPosition[1] = 1.800;
		Event Player.cameraPosition[2] = 3.900;
		Global.zombieKingHealth = 3000;
		Global.zombieSpeed = 145;
		Stop Damage Modification(Event Player.damageMod);
		Start Damage Modification(Event Player, Players On Hero(Hero(정크랫), All Teams), 0, Receivers Damagers and Damage Percent);
		Create Progress Bar HUD Text(Remove From Array(All Players(All Teams), Event Player), Global.zombieKingHealth / 27.500,
			Custom String("{0} {1}", Hero Icon String(Hero(로드호그)), Global.hostZombie), Top, 1, Color(Red), Color(흰색),
			Visible To Values and Color, Default Visibility);
		Set Move Speed(Event Player, Global.zombieSpeed);
		Set Jump Vertical Speed(Event Player, 150);
		Set Damage Dealt(Event Player, 250);
		Set Healing Received(Event Player, 0);
	}
}
